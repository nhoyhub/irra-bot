<script>
    const API_BASE_URL = 'http://127.0.0.1:5000'; // Change to your server URL
    
    // --- MOBILE SIDEBAR TOGGLE ---
    function toggleSidebar() {
        const sidebar = document.getElementById('sidebar');
        const backdrop = document.getElementById('mobileBackdrop');
        const isClosed = sidebar.classList.contains('-translate-x-full');
        
        if (isClosed) {
            sidebar.classList.remove('-translate-x-full');
            backdrop.classList.remove('hidden');
            document.body.style.overflow = 'hidden'; // Prevent scrolling
        } else {
            sidebar.classList.add('-translate-x-full');
            backdrop.classList.add('hidden');
            document.body.style.overflow = 'auto';
        }
    }

    // --- DATA VARIABLES ---
    let allOrdersData = [];
    let filteredOrders = [];
    let currentPage = 1;
    const ITEMS_PER_PAGE = 10;
    let currentOrderKey = null;

    // --- FETCH DATA ---
    async function fetchOrders() {
        const spinner = document.getElementById('loadingSpinner');
        const tbody = document.getElementById('orderTableBody');
        
        spinner.classList.remove('hidden');
        tbody.innerHTML = '';
        
        try {
            const response = await fetch(`${API_BASE_URL}/admin/orders`);
            const data = await response.json();
            
            // Sort by date (Newest first)
            allOrdersData = Object.values(data).sort((a, b) => new Date(b.save_time) - new Date(a.save_time));
            filteredOrders = allOrdersData;
            
            calculateStats();
            renderTable();
            showToast("Dashboard updated!", "success");
        } catch (error) {
            console.error(error);
            tbody.innerHTML = `<tr><td colspan="6" class="text-center py-6 text-red-500">Error connecting to server</td></tr>`;
            showToast("Connection failed", "error");
        } finally {
            spinner.classList.add('hidden');
        }
    }

    // --- RENDER TABLE ---
    function renderTable() {
        const tbody = document.getElementById('orderTableBody');
        tbody.innerHTML = '';
        
        const start = (currentPage - 1) * ITEMS_PER_PAGE;
        const end = start + ITEMS_PER_PAGE;
        const pageData = filteredOrders.slice(start, end);

        if(pageData.length === 0) {
            tbody.innerHTML = `<tr><td colspan="6" class="text-center py-8 text-slate-400">No orders found</td></tr>`;
            return;
        }

        pageData.forEach(order => {
            const isLinkPending = !order.link || order.link.includes('yourdomain.com');
            const tr = document.createElement('tr');
            tr.className = 'hover:bg-slate-50 transition';
            
            tr.innerHTML = `
                <td class="px-6 py-4">
                    <div class="flex items-center gap-3">
                        <div class="w-8 h-8 rounded-full bg-slate-200 flex items-center justify-center font-bold text-slate-600 text-xs">
                            ${order.username.charAt(0).toUpperCase()}
                        </div>
                        <div>
                            <p class="font-bold text-slate-800 text-sm">${order.username}</p>
                            <p class="text-xs text-slate-400">ID: ${order.user_id}</p>
                        </div>
                    </div>
                </td>
                <td class="px-6 py-4">
                    <code class="bg-slate-100 px-2 py-1 rounded text-xs text-slate-600">${order.udid.substring(0, 8)}...</code>
                </td>
                <td class="px-6 py-4 font-bold text-slate-700">$${order.payment_option}</td>
                <td class="px-6 py-4">
                    <span class="px-2 py-1 rounded-full text-xs font-bold ${isLinkPending ? 'bg-amber-100 text-amber-600' : 'bg-green-100 text-green-600'}">
                        ${isLinkPending ? 'Pending' : 'Completed'}
                    </span>
                </td>
                <td class="px-6 py-4">
                     <span class="text-xs ${isLinkPending ? 'text-red-400' : 'text-green-500'}">
                        ${isLinkPending ? 'Missing' : 'Active'}
                     </span>
                </td>
                <td class="px-6 py-4 text-right">
                    <div class="flex justify-end gap-2">
                        <button onclick="openLinkModal('${order.order_key}', '${order.link || ''}', '${order.link_secondary || ''}')" 
                            class="p-2 rounded bg-blue-50 text-blue-600 hover:bg-blue-100" title="Edit Link">
                            ‚úèÔ∏è
                        </button>
                        <button onclick="sendLink('${order.user_id}', '${order.order_key}')" 
                            class="p-2 rounded bg-green-50 text-green-600 hover:bg-green-100" title="Send Email">
                            üìß
                        </button>
                         <button onclick="deleteOrder('${order.order_key}')" 
                            class="p-2 rounded bg-red-50 text-red-600 hover:bg-red-100" title="Delete">
                            üóëÔ∏è
                        </button>
                    </div>
                </td>
            `;
            tbody.appendChild(tr);
        });

        // Update Pagination Text
        document.getElementById('pageNumberDisplay').textContent = `Page ${currentPage} of ${Math.ceil(filteredOrders.length / ITEMS_PER_PAGE) || 1}`;
        document.getElementById('backButton').disabled = currentPage === 1;
        document.getElementById('nextButton').disabled = currentPage * ITEMS_PER_PAGE >= filteredOrders.length;
    }

    // --- STATS ---
    function calculateStats() {
        const total = allOrdersData.length;
        const pending = allOrdersData.filter(o => !o.link || o.link.includes('yourdomain')).length;
        const completed = total - pending;
        const revenue = allOrdersData.reduce((acc, curr) => acc + (parseFloat(curr.payment_option) || 0), 0);

        document.getElementById('statTotalOrders').innerText = total;
        document.getElementById('statCompleted').innerText = completed;
        document.getElementById('statPending').innerText = pending;
        document.getElementById('statRevenue').innerText = `$${revenue.toFixed(2)}`;
        document.getElementById('statCompletionRate').innerText = total > 0 ? `${Math.round((completed/total)*100)}% Rate` : '0% Rate';
    }

    // --- ACTIONS ---
    function openLinkModal(key, l1, l2) {
        currentOrderKey = key;
        document.getElementById('modalLinkInput').value = l1 && !l1.includes('yourdomain') ? l1 : '';
        document.getElementById('modalLinkInput2').value = l2 || '';
        document.getElementById('linkModal').classList.remove('hidden');
    }

    function closeModal(id) {
        document.getElementById(id).classList.add('hidden');
    }

    // Save Link
    document.getElementById('modalSaveButton').addEventListener('click', async () => {
        const btn = document.getElementById('modalSaveButton');
        const l1 = document.getElementById('modalLinkInput').value;
        const l2 = document.getElementById('modalLinkInput2').value;

        btn.innerText = "Saving...";
        
        try {
            const res = await fetch(`${API_BASE_URL}/api/v1/update_link/${currentOrderKey}`, {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({ link1: l1, link2: l2 })
            });

            if(res.ok) {
                showToast("Link Saved!", "success");
                // Update local data without re-fetching
                const order = allOrdersData.find(o => o.order_key === currentOrderKey);
                if(order) { order.link = l1; order.link_secondary = l2; }
                renderTable();
                calculateStats();
                closeModal('linkModal');
            } else {
                showToast("Failed to save", "error");
            }
        } catch(e) { showToast("Error saving", "error"); }
        
        btn.innerText = "Save Changes";
    });

    // Send Email
    async function sendLink(uid, key) {
        if(!confirm("Send email to user?")) return;
        const order = allOrdersData.find(o => o.order_key === key);
        
        if(!order.link || order.link.includes('yourdomain')) {
            showToast("Please save a link first!", "error");
            return;
        }

        try {
            const res = await fetch(`${API_BASE_URL}/api/v1/send_link`, {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({ user_id: uid, link_primary: order.link, link_secondary: order.link_secondary })
            });
            const d = await res.json();
            showToast(d.message, res.ok ? "success" : "error");
        } catch(e) { showToast("Network Error", "error"); }
    }

    // Delete Order
    async function deleteOrder(key) {
        if(!confirm("Delete this order permanently?")) return;
        try {
            await fetch(`${API_BASE_URL}/api/v1/delete_order/${key}`, { method: 'DELETE' });
            allOrdersData = allOrdersData.filter(o => o.order_key !== key);
            filteredOrders = filteredOrders.filter(o => o.order_key !== key);
            renderTable();
            calculateStats();
            showToast("Order deleted", "success");
        } catch(e) { showToast("Error deleting", "error"); }
    }

    // Filters
    function filterByStatus(status) {
        document.querySelectorAll('button[id^="btn-"]').forEach(b => {
            b.className = "px-4 py-1.5 rounded-lg text-sm font-medium bg-slate-100 text-slate-600 whitespace-nowrap hover:bg-slate-200";
        });
        const activeBtn = document.getElementById(`btn-${status}`);
        activeBtn.className = "px-4 py-1.5 rounded-lg text-sm font-medium bg-purple-600 text-white whitespace-nowrap shadow-md";

        if(status === 'all') filteredOrders = allOrdersData;
        else if(status === 'pending') filteredOrders = allOrdersData.filter(o => !o.link || o.link.includes('yourdomain'));
        else filteredOrders = allOrdersData.filter(o => o.link && !o.link.includes('yourdomain'));
        
        currentPage = 1;
        renderTable();
    }

    function filterTable() {
        const term = document.getElementById('searchBar').value.toLowerCase();
        filteredOrders = allOrdersData.filter(o => 
            o.username.toLowerCase().includes(term) || o.udid.toLowerCase().includes(term)
        );
        currentPage = 1;
        renderTable();
    }

    // Pagination
    document.getElementById('nextButton').addEventListener('click', () => { currentPage++; renderTable(); });
    document.getElementById('backButton').addEventListener('click', () => { currentPage--; renderTable(); });

    // Toast Helper
    function showToast(msg, type) {
        const div = document.createElement('div');
        div.className = `fixed bottom-5 right-5 px-6 py-3 rounded-xl shadow-2xl text-white font-medium z-[70] transition-all transform translate-y-10 opacity-0 ${type === 'error' ? 'bg-red-500' : 'bg-green-500'}`;
        div.innerText = msg;
        document.body.appendChild(div);
        
        setTimeout(() => { div.classList.remove('translate-y-10', 'opacity-0'); }, 10);
        setTimeout(() => { div.classList.add('opacity-0'); setTimeout(() => div.remove(), 300); }, 3000);
    }
