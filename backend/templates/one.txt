<script>
    // ‚ö†Ô∏è IMPORTANT: Change this URL to your live server when deploying!
    // const API_BASE_URL = 'https://irra-bot.onrender.com';
    const API_BASE_URL = 'http://127.0.0.1:5000'; 
    
    // --- SIDEBAR LOGIC ---
    function toggleSidebar() {
        const sidebar = document.getElementById('sidebar');
        const backdrop = document.getElementById('mobileBackdrop');
        const isClosed = sidebar.classList.contains('-translate-x-full');
        
        if (isClosed) {
            sidebar.classList.remove('-translate-x-full');
            backdrop.classList.remove('hidden');
            document.body.style.overflow = 'hidden';
        } else {
            sidebar.classList.add('-translate-x-full');
            backdrop.classList.add('hidden');
            document.body.style.overflow = 'auto';
        }
    }

    // --- STATE VARIABLES ---
    let allOrdersData = [];
    let filteredOrders = [];
    let currentPage = 1;
    const ITEMS_PER_PAGE = 10;
    let currentOrderKey = null;

    // --- FETCH DATA ---
    async function fetchOrders() {
        const spinner = document.getElementById('loadingSpinner');
        const tbody = document.getElementById('orderTableBody');
        
        spinner.classList.remove('hidden');
        tbody.innerHTML = '';
        
        try {
            const response = await fetch(`${API_BASE_URL}/admin/orders`);
            const data = await response.json();
            
            // Sort by save_time desc (newest first)
            allOrdersData = Object.values(data).sort((a, b) => new Date(b.save_time) - new Date(a.save_time));
            filteredOrders = allOrdersData;
            
            calculateStats();
            renderTable();
            showToast("LOG POSE UPDATED!", "success");
        } catch (error) {
            console.error("Backend Error:", error);
            tbody.innerHTML = `<tr><td colspan="6" class="text-center py-6 text-red-400 pirate-font">Connection Lost! Is Flask running?</td></tr>`;
            showToast("Connection failed", "error");
        } finally {
            spinner.classList.add('hidden');
        }
    }

    // --- RENDER TABLE ---
    function renderTable() {
        const tbody = document.getElementById('orderTableBody');
        tbody.innerHTML = '';
        
        const start = (currentPage - 1) * ITEMS_PER_PAGE;
        const end = start + ITEMS_PER_PAGE;
        const pageData = filteredOrders.slice(start, end);

        if(pageData.length === 0) {
            tbody.innerHTML = `<tr><td colspan="6" class="text-center py-8 text-slate-500 pirate-font">NO PIRATES FOUND ON RADAR</td></tr>`;
            return;
        }

        pageData.forEach(order => {
            // Check if Primary Link is set
            const hasLink1 = order.link_primary && !order.link_primary.includes('yourdomain.com') && order.link_primary.trim() !== '';
            const hasLink2 = order.link_secondary && order.link_secondary.trim() !== '';

            // Formatting Price
            let priceDisplay = "0.00";
            if (!isNaN(parseFloat(order.payment_option))) {
                priceDisplay = parseFloat(order.payment_option).toLocaleString('en-US', {minimumFractionDigits: 2});
            } else {
                priceDisplay = order.payment_option; // Show text if not a number
            }

            const tr = document.createElement('tr');
            tr.className = 'hover:bg-black/5 transition'; 
            
            tr.innerHTML = `
                <td class="px-6 py-4">
                    <div class="flex items-center gap-3">
                        <div class="w-8 h-8 rounded-full bg-slate-800 border-2 border-yellow-500/50 flex items-center justify-center font-bold text-yellow-500 text-xs shadow-inner shadow-yellow-500/20">
                            ${(order.username || 'U').charAt(0).toUpperCase()}
                        </div>
                        <div>
                            <p class="font-bold text-white text-sm tracking-wide">${order.username || 'Unknown'}</p>
                            <p class="text-xs text-slate-500 font-mono">ID: ${order.user_id}</p>
                        </div>
                    </div>
                </td>
                <td class="px-6 py-4">
                    <code class="bg-black/40 px-2 py-1 rounded text-xs text-slate-400 border border-white/20 font-mono tracking-normal break-all">${order.udid}</code>
                </td>
                <td class="px-6 py-4 font-bold text-yellow-400 font-mono">$${priceDisplay}</td>
                <td class="px-6 py-4">
                    <span class="px-2 py-1 rounded-full text-xs font-bold uppercase ${!hasLink1 ? 'bg-yellow-500/10 text-yellow-500 border border-yellow-500/30' : 'bg-green-500/10 text-green-400 border border-green-500/30'}">
                        ${!hasLink1 ? 'WANTED' : 'CAPTURED'}
                    </span>
                </td>
                <td class="px-6 py-4">
                    <div class="flex flex-col gap-1">
                        <span class="text-xs ${hasLink1 ? 'text-blue-400' : 'text-red-400'}">
                           ${hasLink1 ? '‚úì Esign Ready' : '‚Ä¢ Missing Esign'}
                        </span>
                        ${hasLink2 ? '<span class="text-xs text-green-400">‚úì Cert Ready</span>' : ''}
                    </div>
                </td>
                <td class="px-6 py-4 text-right">

    // --- CALCULATE STATS ---
    function calculateStats() {
        const total = allOrdersData.length;
        const pending = allOrdersData.filter(o => !o.link_primary || o.link_primary.includes('yourdomain') || o.link_primary.trim() === '').length;
        const completed = total - pending;
        
        const revenue = allOrdersData.reduce((acc, curr) => {
            const val = parseFloat(curr.payment_option);
            return acc + (isNaN(val) ? 0 : val);
        }, 0);

        document.getElementById('statTotalOrders').innerText = total;
        document.getElementById('statCompleted').innerText = completed;
        document.getElementById('statPending').innerText = pending;
        document.getElementById('statRevenue').innerText = `$${revenue.toLocaleString('en-US', {minimumFractionDigits: 2})}`;
        document.getElementById('statCompletionRate').innerText = total > 0 ? `${Math.round((completed/total)*100)}% Rate` : '0% Rate';
    }

    // --- MODAL ACTIONS ---
    function openLinkModal(key, l1, l2) {
        currentOrderKey = key;
        // If l1 is the default dummy link, show empty
        document.getElementById('modalLinkInput').value = l1 && !l1.includes('yourdomain') ? l1 : '';
        document.getElementById('modalLinkInput2').value = l2 || '';
        document.getElementById('linkModal').classList.remove('hidden');
    }

    function closeModal(id) {
        document.getElementById(id).classList.add('hidden');
    }

    // SAVE LINKS
    document.getElementById('modalSaveButton').addEventListener('click', async () => {
        const btn = document.getElementById('modalSaveButton');
        const l1 = document.getElementById('modalLinkInput').value;
        const l2 = document.getElementById('modalLinkInput2').value;

        btn.innerText = "Placing...";
        
        try {
            const res = await fetch(`${API_BASE_URL}/api/v1/update_link/${currentOrderKey}`, {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({ link1: l1, link2: l2 })
            });

            if(res.ok) {
                showToast("Course Plotted!", "success");
                // Update local data optimistically
                const order = allOrdersData.find(o => o.order_key === currentOrderKey);
                if(order) { 
                    order.link_primary = l1; 
                    order.link_secondary = l2; 
                    order.link = l1; 
                }
                renderTable();
                calculateStats();
                closeModal('linkModal');
            } else {
                showToast("Failed to save", "error");
            }
        } catch(e) { 
            console.error(e);
            showToast("Network issue", "error"); 
        } finally {
            btn.innerText = "Save Course";
        }
    });

    // SEND MESSAGE (Updated to use new Backend API)
    async function sendLink(uid, key) {
        if(!confirm("Send Message to User?")) return;
        const order = allOrdersData.find(o => o.order_key === key);
        
        // Validation: Must have at least primary link
        if(!order || !order.link_primary || order.link_primary.includes('yourdomain') || order.link_primary.trim() === '') {
            showToast("Please save the Esign link first!", "error");
            return;
        }

        try {
            const res = await fetch(`${API_BASE_URL}/api/v1/send_link`, {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({ 
                    user_id: uid, 
                    link_primary: order.link_primary, 
                    link_secondary: order.link_secondary 
                })
            });
            
            const d = await res.json();
            if(res.ok) {
                showToast("Message Sent!", "success");
            } else {
                showToast("Telegram Error: " + d.message, "error");
            }

        } catch(e) { 
            console.error(e);
            showToast("Network Error", "error"); 
        }
    }

    // DELETE ORDER
    async function deleteOrder(key) {
        if(!confirm("Banish this pirate to the plank permanently?")) return;
        
        try {
            await fetch(`${API_BASE_URL}/api/v1/delete_order/${key}`, { method: 'DELETE' });
            
            allOrdersData = allOrdersData.filter(o => o.order_key !== key);
            filteredOrders = filteredOrders.filter(o => o.order_key !== key);
            renderTable();
            calculateStats();
            showToast("Bounty removed", "success");
        } catch(e) { 
            console.error(e);
            showToast("Network issue", "error"); 
        }
    }

    function filterTable() {
        const term = document.getElementById('searchBar').value.toLowerCase();
        
        // 1. Filter by term
        const tempFiltered = allOrdersData.filter(o => 
            (o.username && o.username.toLowerCase().includes(term)) || 
            (o.udid && o.udid.toLowerCase().includes(term))
        );
        
        // 2. Apply current Status Tab
        const currentStatus = document.querySelector('button[id^="btn-"][class*="bg-gradient-to-r"]').id.replace('btn-', '');
        
        if(currentStatus === 'all') filteredOrders = tempFiltered;
        else if(currentStatus === 'pending') filteredOrders = tempFiltered.filter(o => !o.link_primary || o.link_primary.includes('yourdomain'));
        else filteredOrders = tempFiltered.filter(o => o.link_primary && !o.link_primary.includes('yourdomain'));
        
        currentPage = 1;
        renderTable();
    }
